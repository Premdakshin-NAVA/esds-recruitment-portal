-- ================================================ -- TALENT NEXUS — Supabase Schema -- Run this in Supabase SQL Editor (all at once) -- ================================================ -- 1. USER PROFILES (extends Supabase auth.users) create table public.profiles ( id uuid references auth.users(id) on delete cascade primary key, full_name text not null, email text not null, role text not null check (role in ('admin','ta','hm','developer')), is_active boolean default true, created_at timestamptz default now() ); -- 2. TEAM MEMBERS create table public.team_members ( id uuid default gen_random_uuid() primary key, type text not null check (type in ('TA','HM')), full_name text not null, email text, added_by uuid references public.profiles(id), created_at timestamptz default now() ); -- 3. CANDIDATES create table public.candidates ( id uuid default gen_random_uuid() primary key, name text not null, email text, phone text, org text, desig text, exp numeric, rexp numeric, edu text, np text, np_neg boolean default false, cloc text, ploc text, wfo text, shift text, src text, slots text, curctc numeric, ctc_fixed numeric, ctc_variable numeric, expctc_min numeric, expctc_max numeric, expctc_neg boolean default false, created_by uuid references public.profiles(id), created_at timestamptz default now() ); -- 4. PROCESSES create table public.processes ( id uuid default gen_random_uuid() primary key, candidate_id uuid references public.candidates(id) on delete cascade, role text not null, dept text, etype text default 'Full time', jloc text, rounds integer default 2, assigned_ta uuid references public.profiles(id), current_hm_name text, stage text default 'Screening call completed', status text default 'active' check (status in ('active','scheduled','hold','cancelled','closed-selected','closed-rejected')), sal_visible boolean default true, days integer default 0, created_by uuid references public.profiles(id), created_at timestamptz default now(), updated_at timestamptz default now() ); -- 5. PROCESS ROUNDS create table public.process_rounds ( id uuid default gen_random_uuid() primary key, process_id uuid references public.processes(id) on delete cascade, round_number integer not null, round_name text not null, scheduled_at timestamptz, outcome text check (outcome in ('selected','rejected','noshow','pending')), created_at timestamptz default now() ); -- 6. ROUND INTERVIEWERS (supports multiple HMs per round) create table public.round_interviewers ( id uuid default gen_random_uuid() primary key, round_id uuid references public.process_rounds(id) on delete cascade, hm_id uuid references public.profiles(id), hm_name text not null, confirmed boolean default false, created_at timestamptz default now() ); -- 7. TEF RATINGS (one row per HM per round) create table public.tef_ratings ( id uuid default gen_random_uuid() primary key, round_id uuid references public.process_rounds(id) on delete cascade, hm_id uuid references public.profiles(id), hm_name text, c1 integer check (c1 between 1 and 5), -- Passion and Motivation c2 integer check (c2 between 1 and 5), -- Commitment and Enthusiasm c3 integer check (c3 between 1 and 5), -- Growth and Continuous Improvement c4 integer check (c4 between 1 and 5), -- Adaptability and Resilience c5 integer check (c5 between 1 and 5), -- Collaboration and Communication c6 integer check (c6 between 1 and 5), -- Leadership and Accountability c7 integer check (c7 between 1 and 5), -- Empathy and Humility c8 integer check (c8 between 1 and 5), -- Strategic and Analytical Thinking c9 integer check (c9 between 1 and 5), -- Problem Solving and Conflict Management c10 integer check (c10 between 1 and 5), -- Integrity and Risk-Taking comments text, hire_reason text, no_hire_reason text, submitted_at timestamptz default now() ); -- 8. TEF HR SECTION create table public.tef_hr ( id uuid default gen_random_uuid() primary key, process_id uuid references public.processes(id) on delete cascade, hr_id uuid references public.profiles(id), strengths text, weaknesses text, overall_observation text, reason_to_hire text, designation text, grade text, reporting text, department text, location text, recommendation boolean, submitted_at timestamptz default now() ); -- 9. TIMELINE EVENTS create table public.timeline_events ( id uuid default gen_random_uuid() primary key, process_id uuid references public.processes(id) on delete cascade, label text not null, detail text, conductor_name text, updated_by uuid references public.profiles(id), updated_by_name text, state text default 'active' check (state in ('done','active','hold','pend')), created_at timestamptz default now() ); -- 10. COMMENTS create table public.comments ( id uuid default gen_random_uuid() primary key, process_id uuid references public.processes(id) on delete cascade, by_id uuid references public.profiles(id), by_name text, text text not null, created_at timestamptz default now() );
